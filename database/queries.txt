________________________________________________________________________________
                             Λειτουργία συστήματος
________________________________________________________________________________

a. {baseURL}/questionnaire/:questionnaireID
--------------------------------------------
select questionnaireTitle from `questionnaire` where questionnaireID = [questionnaireID];

select keyword from `keywords` where questionnaireID = [questionnaireID];

select qID, qtext, required, type from `questions` where questionnaireID = [questionnaireID] order by qID;
________________________________________________________________________________

b. {baseURL}/question/:questionnaireID/:questionID
---------------------------------------------------
select qtext, required, type from `questions` where questionnaireID = [questionnaireID] and qID = [questionID];
select optID, opttxt, nextqID from `options` where questionnaireID = [questionnaireID] and qID = [questionID] order by optID;
________________________________________________________________________________

c. {baseURL}/doanswer/:questionnaireID/:questionID/:session/:optionID
----------------------------------------------------------------------
insert into `answers` values ([optionID],[questionnaireID],[session],[questionID],current_timestamp(),[ans_str]);
________________________________________________________________________________

d. {baseURL}/getsessionanswers/:questionnaireID/:session
---------------------------------------------------------
select qID, ans, ans_str from `answers` where questionnaireID = [questionnaireID] and session = [session] order by qID;
________________________________________________________________________________

e. {baseURL}/getquestionanswers/:questionnaireID/:questionID
-------------------------------------------------------------
select session, ans_str from `answers` where questionnaireID = [questionnaireID] and qID = [questionID] order by ans_datetime desc;
select session, ans from `answers` where questionnaireID = [questionnaireID] and qID = [questionID] order by ans_datetime desc;
